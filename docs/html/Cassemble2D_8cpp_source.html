312 int retriangulate(double * x_center, double * TE, double sqdelta, double * out_reTriangle_list, int is_placePointOnCap){
313  // C Variables and Arrays.
314  int i=0, k=0, edgdx0=0, edgdx1=0, Rdx=0;
315  double v=0, lam1=0, lam2=0, term1=0, term2=0;
316  double nu_a[2], nu_b[2], nu_c[2]; // Normals
317  double p[2];
318  double q[2];
319  double a[2];
320  double b[2];
321  double y1[2];
322  double y2[2];
323  double orientation;
324 
325  bool is_onEdge=false, is_firstPointLiesOnVertex=true;
326  // The upper bound for the number of required points is 9
327  // Hence 9*3 is an upper bound to encode all resulting triangles
328  // Hence we can hardcode how much space needs to bee allocated
329  // (This upper bound is thight! Check Christian Vollmann's thesis for more information.)
330 
331  double R[9*2]; // Vector containing all intersection points.
332  doubleVec_tozero(R, 9*2);
333 
334  // Compute Normals of the Triangle
335  orientation = -signDet(TE);
336  rightNormal(&TE[0], &TE[2], orientation, nu_a);
337  rightNormal(&TE[2], &TE[4], orientation, nu_b);
338  rightNormal(&TE[4], &TE[0], orientation, nu_c);
339 
340  for (k=0; k<3; k++){
341  edgdx0 = k;
342  edgdx1 = (k+1) % 3;
343 
344  doubleVec_copyTo(&TE[2*edgdx0], p, 2);
345  doubleVec_copyTo(&TE[2*edgdx1], q, 2);
346  doubleVec_subtract(q, x_center, a, 2);
347  doubleVec_subtract(p, q, b, 2);
348 
349  if (vec_sqL2dist(p, x_center, 2) <= sqdelta){
350  doubleVec_copyTo(p, &R[2*Rdx], 2);
351  is_onEdge = false; // This point does not lie on the edge.
352  Rdx += 1;
353  }
354  // PQ-Formula to solve quadratic problem
355  v = pow(vec_dot(a, b, 2), 2) - (vec_dot(a, a, 2) - sqdelta) * vec_dot(b, b, 2);
356  // If there is no sol to the quadratic problem, there is nothing to do.
357  if (v >= 0){
358  term1 = -vec_dot(a, b, 2) / vec_dot(b, b, 2);
359  term2 = sqrt(v) / vec_dot(b, b, 2);
360  // Vieta's Formula for computing the roots
361  if (term1 > 0){
362  lam1 = term1 + term2;
363  // (vec_dot(a, a, 2) - sqdelta) / vec_dot(b, b, 2) is the q, of the pq-Formula
364  lam2 = 1/lam1*(vec_dot(a, a, 2) - sqdelta) / vec_dot(b, b, 2);
365  } else {
366  lam2 = term1 - term2;
367  lam1 = 1/lam2*(vec_dot(a, a, 2) - sqdelta) / vec_dot(b, b, 2);
368  }
369  //lam1 = term1 + term2;
370  //lam2 = term1 - term2;
371  // Set Intersection Points
372  doubleVec_scale(lam1, b, y1, 2);
373  doubleVec_add(y1, q, y1, 2);
374  doubleVec_scale(lam2, b, y2, 2);
375  doubleVec_add(y2, q, y2, 2);
376 
377  // Check whether the first lambda "lies on the Triangle".
378  if ((0 <= lam1) && (lam1 <= 1)){
379  is_firstPointLiesOnVertex = is_firstPointLiesOnVertex && (bool)Rdx;
380  // Check whether the predecessor lied on the edge
381  if (is_onEdge && is_placePointOnCap){
382  Rdx += placePointOnCap(&R[2*(Rdx-1)], y1, x_center, sqdelta, TE, nu_a, nu_b, nu_c, orientation, Rdx, R);
383  }
384  // Append y1
385  doubleVec_copyTo(y1, &R[2*Rdx], 2);
386  is_onEdge = true; // This point lies on the edge.
387  Rdx += 1;
388  }
389  // Check whether the second lambda "lies on the Triangle".
390  if ((0 <= lam2) && (lam2 <= 1) && (scal_sqL2dist(lam1, lam2) > 0)){
391  is_firstPointLiesOnVertex = is_firstPointLiesOnVertex && (bool)Rdx;
392 
393 
394  // Check whether the predecessor lied on the edge
395  if (is_onEdge && is_placePointOnCap){
396  Rdx += placePointOnCap(&R[2*(Rdx-1)], y2, x_center, sqdelta, TE, nu_a, nu_b, nu_c, orientation, Rdx, R);
397  }
398 
399  // Append y2
400  doubleVec_copyTo(y2, &R[2*Rdx], 2);
401  is_onEdge = true; // This point lies on the edge.
402  Rdx += 1;
403  }
404  }
405  }
406  if (is_onEdge && (!is_firstPointLiesOnVertex && Rdx > 1) && is_placePointOnCap){
407  Rdx += placePointOnCap(&R[2*(Rdx-1)], &R[0], x_center, sqdelta, TE, nu_a, nu_b, nu_c, orientation, Rdx, R);
408  }
409 
410  // Construct List of Triangles from intersection points
411  if (Rdx < 3){
412  // In this case the content of the array out_RE will not be touched.
413  return 0;
414  } else {
415 
416  for (k=0; k < (Rdx - 2); k++){
417  for (i=0; i<2; i++){
418  // i is the index which runs first, then h (which does not exist here), then k
419  // hence if we increase i, the *-index (of the pointer) inreases in the same way.
420  // if we increase k, there is quite a 'jump'
421  out_reTriangle_list[2 * (3 * k + 0) + i] = R[i];
422  out_reTriangle_list[2 * (3 * k + 1) + i] = R[2*(k + 1) + i];
423  out_reTriangle_list[2 * (3 * k + 2) + i] = R[2*(k + 2) + i];
424  }
425  }
426  // Excessing the bound out_Rdx will not lead to an error but simply to corrupted data!
427 
428  return Rdx - 2; // So that, it acutally contains the number of triangles in the retriangulation
429  }
430 }

855 void solve2x2(double * A, double * b, double * x){
856  int dx0 = 0, dx1 = 1;
857  double l=0, u=0;
858 
859  // Column Pivot Strategy
860  if (absolute(A[0]) < absolute(A[2])){
861  dx0 = 1;
862  dx1 = 0;
863  }
864 
865  // Check invertibility
866  if (A[2*dx0] == 0){
867  // raise LinAlgError("in solve2x2. Matrix not invertible.")
868  cout << "in solve2x2. Matrix not invertible." << endl;
869  abort();
870  }
871 
872  // LU Decomposition
873  l = A[2*dx1]/A[2*dx0];
874  u = A[2*dx1+1] - l*A[2*dx0+1];
875 
876  // Check invertibility
877  if (u == 0){
878  // raise LinAlgError("in solve2x2. Matrix not invertible.")
879  cout << "in solve2x2. Matrix not invertible." << endl;
880  abort();
881  }
882 
883  // LU Solve
884  x[1] = (b[dx1] - l*b[dx0])/u;
885  x[0] = (b[dx0] - A[2*dx0+1]*x[1])/A[2*dx0];
886  return;
887 }
